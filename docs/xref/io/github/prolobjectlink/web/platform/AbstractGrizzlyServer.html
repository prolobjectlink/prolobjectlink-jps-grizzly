93  	public AbstractGrizzlyServer(int serverPort) {
94  		super(serverPort);
95  		try {
96  
97  			databaseServer.startup();
98  
99  			InetAddress localHost = InetAddress.getLocalHost();
100 			String localHostAddr = localHost.getHostAddress();
101 			// PortRange portRange = new PortRange(serverPort, 9220);
102 			NetworkListener localHostListener = new NetworkListener("localhost", localHostAddr, serverPort);
103 			InetAddress loopback = InetAddress.getLoopbackAddress();
104 			String loopbackAddr = loopback.getHostAddress();
105 			NetworkListener loopbackListener = new NetworkListener("loopback", loopbackAddr, serverPort);
106 			URI baseUri = URI.create("http://" + localHostAddr + ":" + serverPort);
107 			httpserver = GrizzlyHttpServerFactory.createHttpServer(baseUri);
108 			httpserver.addListener(localHostListener);
109 			httpserver.addListener(loopbackListener);
110 			
111 			// Web context initialization
112 			GrizzlyWebContext context = new GrizzlyWebContext();
113 
114 			// Manager Servlets Initialization
115 			RootServlet root = new RootServlet();
116 			context.addServlet(root.getClass().getName(), root).addMapping("/");
117 			WelcomeServlet home = new WelcomeServlet();
118 			context.addServlet(home.getClass().getName(), home).addMapping("/welcome");
119 			NotFoundServlet notFoundServlet = new NotFoundServlet();
120 			context.addServlet(notFoundServlet.getClass().getName(), notFoundServlet).addMapping("/notfound");
121 			ApplicationServlet app = new ApplicationServlet();
122 			context.addServlet(app.getClass().getName(), app).addMapping("/pas/applications");
123 			DatabaseServlet db = new DatabaseServlet();
124 			context.addServlet(db.getClass().getName(), db).addMapping("/pas/databases");
125 			ModelServlet m = new ModelServlet();
126 			context.addServlet(m.getClass().getName(), m).addMapping("/pas/models");
127 			ConfigurationServlet etc = new ConfigurationServlet();
128 			context.addServlet(etc.getClass().getName(), etc).addMapping("/pas/configuration");
129 			ConnectorsServlet conn = new ConnectorsServlet();
130 			context.addServlet(conn.getClass().getName(), conn).addMapping("/pas/connector");
131 			PersistenceServlet pers = new PersistenceServlet();
132 			context.addServlet(pers.getClass().getName(), pers).addMapping("/pas/persistence");
133 			ProviderServlet pro = new ProviderServlet();
134 			context.addServlet(pro.getClass().getName(), pro).addMapping("/pas/provider");
135 			LogsServlet logs = new LogsServlet();
136 			context.addServlet(logs.getClass().getName(), logs).addMapping("/pas/logs");
137 			ManagerServlet man = new ManagerServlet(databaseServer, this);
138 			context.addServlet(man.getClass().getName(), man).addMapping("/pas/manager");
139 			DocumentsServlet doc = new DocumentsServlet();
140 			context.addServlet(doc.getClass().getName(), doc).addMapping("/doc");
141 			ProlobjectlinkServlet pas = new ProlobjectlinkServlet(databaseServer, this);
142 			context.addServlet(pas.getClass().getName(), pas).addMapping("/pas/information");
143 			AboutServlet about = new AboutServlet();
144 			context.addServlet(about.getClass().getName(), about).addMapping("/pas/about");
145 			UserServlet users = new UserServlet();
146 			context.addServlet(users.getClass().getName(), users).addMapping("/pas/users");
147 
148 			// operations on manager
149 			DeleteApplicationServlet delapp = new DeleteApplicationServlet();
150 			context.addServlet(delapp.getClass().getName(), delapp).addMapping("/pas/application/delete/*");
151 			DeleteDatabaseServlet deldb = new DeleteDatabaseServlet();
152 			context.addServlet(deldb.getClass().getName(), deldb).addMapping("/pas/database/delete/*");
153 			ExportApplicationServlet exapp = new ExportApplicationServlet();
154 			context.addServlet(exapp.getClass().getName(), exapp).addMapping("/pas/application/export/*");
155 			ExportDatabaseServlet exdb = new ExportDatabaseServlet();
156 			context.addServlet(exdb.getClass().getName(), exdb).addMapping("/pas/database/export/*");
157 			UploadApplicationServlet uploadapp = new UploadApplicationServlet();
158 			context.addServlet(uploadapp.getClass().getName(), uploadapp).addMapping("/pas/uploadapp/*");
159 			UploadDatabaseServlet uploaddb = new UploadDatabaseServlet();
160 			context.addServlet(uploaddb.getClass().getName(), uploaddb).addMapping("/pas/uploaddb/*");
161 			InformationServlet showdb = new InformationServlet();
162 			context.addServlet(showdb.getClass().getName(), showdb).addMapping("/pas/database/show/*");
163 			LogoutServlet logout = new LogoutServlet();
164 			context.addServlet(logout.getClass().getName(), logout).addMapping("/pas/logout");
165 
166 			// server misc
167 			ResourceLoader miscGenerator = new GrizzlyMiscellaneousLoader();
168 			List<ServletUrlMapping> miscPaths = miscGenerator.getMappings();
169 			for (ServletUrlMapping servletUrlMapping : miscPaths) {
170 				Servlet servlet = servletUrlMapping.getServlet();
171 				context.addServlet(servletUrlMapping.getMappingUrl(), servlet)
172 						.addMapping(servletUrlMapping.getMappingUrl());
173 			}
174 
175 			// server resources
176 			ResourceLoader resourcesGenerator = new GrizzlyResourceLoader();
177 			List<ServletUrlMapping> resourcespaths = resourcesGenerator.getMappings();
178 			for (ServletUrlMapping servletUrlMapping : resourcespaths) {
179 				Servlet servlet = servletUrlMapping.getServlet();
180 				context.addServlet(servletUrlMapping.getMappingUrl(), servlet)
181 						.addMapping(servletUrlMapping.getMappingUrl());
182 			}
183 
184 			// server resources
185 			ResourceLoader documentsGenerator = new GrizzlyResourceLoader();
186 			List<ServletUrlMapping> documentspaths = documentsGenerator.getMappings();
187 			for (ServletUrlMapping servletUrlMapping : documentspaths) {
188 				Servlet servlet = servletUrlMapping.getServlet();
189 				context.addServlet(servletUrlMapping.getMappingUrl(), servlet)
190 						.addMapping(servletUrlMapping.getMappingUrl());
191 			}
192 
193 			// applications models
194 			ModelGenerator modelGenerator = new GrizzlyModelGenerator();
195 			try {
196 				List<PersistenceUnitInfo> units = modelGenerator.getPersistenceUnits();
197 				for (PersistenceUnitInfo unit : units) {
198 					DatabaseDriver databaseDriver = DatabaseDriverFactory.createDriver(unit);
199 					if (!databaseDriver.getDatabasePing()) {
200 						databaseDriver.createDatabase();
201 						JPAPersistenceUnitInfo jpaUnit = (JPAPersistenceUnitInfo) unit;
202 						String name = jpaUnit.getPersistenceProviderClassName();
203 						Class<?> cls = JavaReflect.classForName(name);
204 						Object object = JavaReflect.newInstance(cls);
205 						PersistenceProvider provider = (PersistenceProvider) object;
206 						provider.generateSchema(unit, unit.getProperties());
207 					}
208 				}
209 			} catch (SQLException e) {
210 				// do nothing
211 			}
212 
213 			// applications controllers
214 			ControllerGenerator controllerGenerator = new GrizzlyControllerGenerator();
215 			List<ServletUrlMapping> mappings = controllerGenerator.getMappings();
216 			for (ServletUrlMapping servletUrlMapping : mappings) {
217 				Servlet servlet = context.createServlet(servletUrlMapping.getServlet().getClass());
218 				context.addServlet(servletUrlMapping.getServlet().getClass().getName(), servlet)
219 						.addMapping(servletUrlMapping.getMappingUrl());
220 			}
221 
222 			// restart servlet
223 			RestartServlet restart = new RestartServlet(modelGenerator, controllerGenerator, context, databaseServer,
224 					this);
225 			context.addServlet(restart.getClass().getName(), restart).addMapping("/pas/restart");
226 
227 			context.deploy(httpserver);
228 
229 		} catch (UnknownHostException e) {
230 			LoggerUtils.error(getClass(), LoggerConstants.UNKNOWN_HOST, e);
231 		} catch (ServletException e) {
232 			LoggerUtils.error(getClass(), LoggerConstants.SERVLET_ERROR, e);
233 		} catch (ClassNotFoundException e) {
234 			LoggerUtils.error(getClass(), LoggerConstants.CLASS_NOT_FOUND, e);
235 		} catch (IOException e) {
236 			LoggerUtils.error(getClass(), LoggerConstants.IO, e);
237 		} catch (Exception e) {
238 			if (!(e instanceof BindException)) {
239 				LoggerUtils.error(getClass(), LoggerConstants.RUNTIME_ERROR, e);
240 			}
241 		}
242 
243 	}
