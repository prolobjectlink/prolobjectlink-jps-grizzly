183 				ModelGenerator modelGenerator = new GrizzlyModelGenerator();
184 				List<PersistenceUnitInfo> units = modelGenerator.getPersistenceUnits();
185 				for (PersistenceUnitInfo unit : units) {
186 					DatabaseDriver databaseDriver = DatabaseDriverFactory.createDriver(unit);
187 					if (!databaseDriver.getDatabasePing()) {
188 						databaseDriver.createDatabase();
189 						JPAPersistenceUnitInfo jpaUnit = (JPAPersistenceUnitInfo) unit;
190 						String name = jpaUnit.getPersistenceProviderClassName();
191 						Class<?> cls = JavaReflect.classForName(name);
192 						Object object = JavaReflect.newInstance(cls);
193 						PersistenceProvider provider = (PersistenceProvider) object;
194 						provider.generateSchema(unit, unit.getProperties());
195 					}
196 				}

204 			for (ServletUrlMapping servletUrlMapping : mappings) {
