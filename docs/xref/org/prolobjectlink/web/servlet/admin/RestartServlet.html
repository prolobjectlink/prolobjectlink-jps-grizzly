28  import javax.script.ScriptEngineManager;
29  import javax.servlet.Servlet;
30  import javax.servlet.ServletException;
31  import javax.servlet.annotation.WebServlet;
32  import javax.servlet.http.HttpServletRequest;
33  import javax.servlet.http.HttpServletResponse;
34  
35  import org.prolobjectlink.logging.LoggerConstants;
36  import org.prolobjectlink.logging.LoggerUtils;
37  import org.prolobjectlink.web.function.AssetFunction;
38  import org.prolobjectlink.web.function.PathFunction;
39  import org.prolobjectlink.web.servlet.AbstractServlet;
40  
41  import io.marioslab.basis.template.Template;
42  import io.marioslab.basis.template.TemplateContext;
43  import io.marioslab.basis.template.TemplateLoader;
44  import io.marioslab.basis.template.TemplateLoader.ClasspathTemplateLoader;
45  
46  @WebServlet
47  public class RestartServlet extends AbstractServlet implements Servlet {
48  
49  	private static final long serialVersionUID = 7313381101418470194L;
50  
51  	@Override
52  	protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
53  
54  		Runtime runtime = Runtime.getRuntime();
55  		try {
56  
57  			String lib = getLibDirectory().getCanonicalPath();
58  			String console = "org.prolobjectlink.db.prolog.jpl.swi.SwiPrologDatabaseConsole";
59  			runtime.exec("java -classpath " + lib + "/* " + console + " -m");
60  			runtime.exec("java -classpath " + lib + "/* " + console + " -j " + lib + "/prolobjectlink-jpx-model.jar");
61  
62  			// reload prolobjectlink-jpx-model.jar
63  			// restart database
64  
65  		} catch (IOException e) {
66  			LoggerUtils.error(getClass(), LoggerConstants.IO, e);
67  		}
68  
69  		// request information
70  		String protocol = req.getScheme();
71  		String host = req.getHeader("host");
72  
73  		// Engine and context
74  		TemplateLoader loader = new ClasspathTemplateLoader();
75  		Template template = loader.load("/org/prolobjectlink/web/html/manager.html");
76  		TemplateContext context = new TemplateContext();
77  
78  		// java virtual machine
79  		context.set("jvm", System.getProperty("java.vm.name"));
80  		context.set("jvendor", System.getProperty("java.vendor"));
81  		context.set("jversion", System.getProperty("java.version"));
82  
83  		// operating system
84  		context.set("osname", System.getProperty("os.name"));
85  		context.set("osversion", System.getProperty("os.version"));
86  		context.set("osarch", System.getProperty("os.arch"));
87  
88  		// prolog engine
89  		ScriptEngineManager manager = new ScriptEngineManager();
90  		ScriptEngine engine = manager.getEngineByName("Prolog");
91  		ScriptEngineFactory factory = engine.getFactory();
92  		context.set("plversion", factory.getParameter(ScriptEngine.ENGINE_VERSION));
93  		context.set("plname", factory.getParameter(ScriptEngine.NAME));
94  
95  		// servlet container
96  		context.set("serverversion", "4.1.1");
97  		context.set("servername", "Grizzly");
98  
99  		// functions
100 		context.set("path", new PathFunction("pas", protocol, host));
101 		context.set("asset", new AssetFunction(protocol, host));
102 
103 		// render
104 		template.render(context, resp.getOutputStream());
105 
106 		// response
107 		resp.setStatus(HttpServletResponse.SC_OK);
108 
109 		// response
110 		resp.setStatus(HttpServletResponse.SC_OK);
111 	}
