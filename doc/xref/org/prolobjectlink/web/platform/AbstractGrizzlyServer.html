242 				for (PersistenceUnitInfo unit : units) {
243 					DatabaseDriver databaseDriver = DatabaseDriverFactory.createDriver(unit);
244 					if (!databaseDriver.getDatabasePing()) {
245 						databaseDriver.createDatabase();
246 						JPAPersistenceUnitInfo jpaUnit = (JPAPersistenceUnitInfo) unit;
247 						String name = jpaUnit.getPersistenceProviderClassName();
248 						Class<?> cls = JavaReflect.classForName(name);
249 						Object object = JavaReflect.newInstance(cls);
